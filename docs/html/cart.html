typescriptsrc/cart/cart.controller.ts
@Controller('cart')
@UseGuards(AuthGuard)
export class CartController {
  constructor(private readonly cartService: CartService) {}

  @Get()
  async getCart(@Req() request: Request & { user: JwtPayloadDto }) {
    return success(await this.cartService.getCart(request.user.userId));
  }

  @Post('items')
  async addItem(@Req() request: Request & { user: JwtPayloadDto }, @Body() dto: AddCartItemDto) {
    return success(await this.cartService.addItem(request.user.userId, dto));
  }
}
